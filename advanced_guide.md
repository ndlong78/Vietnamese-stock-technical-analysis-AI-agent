import pandas as pd
import matplotlib.pyplot as plt

class PerformanceDashboard:
    def __init__(self, journal_file='trading_journal.csv'):

    def overall_stats(self):
        """Thá»‘ng kÃª tá»•ng quan"""
        total_trades = len(self.df)
        wins = len(self.df[self.df['Actual'] == 'Win'])
        losses = len(self.df[self.df['Actual'] == 'Loss'])
        
        win_rate = (wins / total_trades) * 100 if total_trades > 0 else 0
        
        total_pnl = self.df['PnL'].sum()
        avg_win = self.df[self.df['Actual'] == 'Win']['PnL'].mean()
        avg_loss = self.df[self.df['Actual'] == 'Loss']['PnL'].mean()
        
        return {
            'total_trades': total_trades,
            'wins': wins,
            'losses': losses,
            'win_rate': f"{win_rate:.1f}%",
            'total_pnl': f"{total_pnl:,}Ä‘",
            'avg_win': f"{avg_win:,}Ä‘",
            'avg_loss': f"{avg_loss:,}Ä‘",

        ai_wins = len(ai_trades[ai_trades['Actual'] == 'Win'])

        ai_pnl = ai_trades['PnL'].sum()
        
        # So sÃ¡nh confidence vs actual
        high_conf = ai_trades[ai_trades['AIConfidence'] >= 75]

            'ai_pnl': f"{ai_pnl:,}Ä‘",
            'high_conf_trades': len(high_conf),
            'high_conf_win_rate': f"{high_conf_rate:.1f}%"
        }

        return strategy_stats
